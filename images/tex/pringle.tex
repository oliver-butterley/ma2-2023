\documentclass{standalone}
\usepackage{pgfplots}
\pgfplotsset{compat=1.13}
\usepackage{amsmath}

\begin{document}

\begin{tikzpicture}
    \begin{axis}[
            view={55}{30},
            axis lines=center,
            xlabel=\(x\), ylabel=\(y\),
            xmin=-1, xmax=1,
            ymin=-1, ymax=1,
            zmin=-1, zmax=1,
        ]
        \addplot3[
            surf,
            samples=31,
            opacity = 0.5,
            domain=-1:0,
            y domain=-1:1,
            z buffer=auto,
        ]
        {x^2-y^2};
    \end{axis}
    \begin{axis}[
            view={55}{30},
            axis lines=center,
            axis x line = none,
            xmin=-1, xmax=1,
            ymin=-1, ymax=1,
            zmin=-1, zmax=1,
        ]
        \addplot3[
            surf,
            samples=31,
            opacity = 0.5,
            domain=0:1,
            y domain=-1:1,
            z buffer=auto,
        ]
        {x^2-y^2};
    \end{axis}
